% ==============================
% 统一图片插入命令
% ==============================

% 用于插入 TikZ / PlantUML 等 LaTeX 源码图。
% 参数：
% #1 最大宽度，默认 0.86\textwidth
% #2 最大高度，默认 0.58\textheight
% #3 图源码路径
% #4 图标题
% #5 图标签
\NewDocumentCommand{\WhuDiagramFigure}{O{0.86\textwidth} O{0.58\textheight} m m m}{%
  \begin{figure}[H]
    \centering
    \begin{adjustbox}{max totalsize={#1}{#2},center}
      \input{#3}
    \end{adjustbox}
    \caption{#4}
    \label{#5}
  \end{figure}
}

% 用于插入 PNG / JPG / PDF 等图片文件。
% 参数：
% #1 最大宽度，默认 0.86\textwidth
% #2 最大高度，默认 0.58\textheight
% #3 图片路径
% #4 图标题
% #5 图标签
\NewDocumentCommand{\WhuImageFigure}{O{0.86\textwidth} O{0.58\textheight} m m m}{%
  \begin{figure}[H]
    \centering
    \begin{adjustbox}{max totalsize={#1}{#2},center}
      \includegraphics{#3}
    \end{adjustbox}
    \caption{#4}
    \label{#5}
  \end{figure}
}

% 智能占位图：
% 如果图片存在则插入图片；如果不存在则显示占位框。
\newcommand{\WhuAutoFigure}[5]{%
  \begin{figure}[H]
    \centering
    \IfFileExists{#3}{%
      \includegraphics[width=#1,height=#2,keepaspectratio]{#3}%
    }{%
      \fbox{%
        \parbox[c][#2][c]{#1}{%
          \centering
          \vspace{6pt}
          {\bfseries 占位图}\\[6pt]
          #4\\[8pt]
          \texttt{\detokenize{#3}}\\[6pt]
          将真实图片放到该路径并重新编译即可
          \vspace{6pt}
        }%
      }%
    }%
    \caption{#4}
    \label{#5}
  \end{figure}
}
