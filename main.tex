% ===== 智能占位图开始 =====
% 用法：
% \WhuAutoFigure{宽度}{高度}{图片路径}{图题}{label}
% 如果图片存在：自动插入图片；
% 如果图片不存在：显示占位框。
\newcommand{\WhuAutoFigure}[5]{%
  \begin{figure}[H]
    \centering
    \IfFileExists{#3}{%
      \includegraphics[width=#1,height=#2,keepaspectratio]{#3}%
    }{%
      \fbox{%
        \parbox[c][#2][c]{#1}{%
          \centering
          \vspace{6pt}
          {\bfseries 占位图}\\[6pt]
          #4\\[8pt]
          \texttt{\detokenize{#3}}\\[6pt]
          将真实图片放到该路径并重新编译即可
          \vspace{6pt}
        }%
      }%
    }%
    \caption{#4}
    \label{#5}
  \end{figure}
}
