    \end{adjustbox}
    \caption{#4}
    \label{#5}
  \end{figure}
}

% ==============================
% 统一字段表命令
% ==============================

% 用于数据库字段表、接口字段表等三列表格。
% 参数：
% #1 第一列宽度，默认 0.25\textwidth
% #2 第二列宽度，默认 0.22\textwidth
% #3 第三列宽度，默认 0.43\textwidth
% #4 表题
% #5 表标签
% #6 表格内容
\NewDocumentCommand{\WhuFieldTable}{O{0.25\textwidth} O{0.22\textwidth} O{0.43\textwidth} m m +m}{%
  \begingroup
  \zihao{5}
  \begin{longtable}{L{#1} L{#2} L{#3}}
    \caption{#4}\label{#5}\\
    \toprule
    字段名 & 类型 & 说明 \\
    \midrule
    #6
    \bottomrule
  \end{longtable}
  \endgroup
}

% ==============================

% #3 第三列宽度，默认 0.33\textwidth
% #4 第四列宽度，默认 0.30\textwidth
% #5 表题
% #6 表标签
% #7 表格内容
\NewDocumentCommand{\WhuInterfaceTable}{O{0.13\textwidth} O{0.12\textwidth} O{0.31\textwidth} O{0.28\textwidth} m m +m}{%

  \zihao{5}
  \setlength{\tabcolsep}{3pt}
  \begin{longtable}{L{#1} L{#2} L{#3} L{#4}}
    \caption{#5}\label{#6}\\
    \toprule
    模块 & 方法 & 接口路径 & 说明 \\
    \midrule
    #7
    \bottomrule
  \end{longtable}
  \endgroup
}
